function encontrarPalabraMasLarga(lista) { let masLarga = ""; // Variable local para el récord for (let i = 0; i < lista.length; i++) { if (lista[i].length > masLarga.length) { masLarga = lista[i]; } } return masLarga; // "Devolvemos" el resultado final } // Así la usamos: let frutas = ["apple", "banana", "kiwi", "watermelon"]; let resultado = encontrarPalabraMasLarga(frutas); alert("La más larga es: " + resultado);
